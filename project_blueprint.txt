from langchain_google_genai import GoogleGenAIEmbeddings
from langchain_community.vectorstores import Chroma
from langchain_text_splitters import RecursiveCharacterTextSplitter

# Initialize Gemini Embeddings Engine for converting text blocks into mathematical vectors
embeddings = GoogleGenAIEmbeddings(model="models/embedding-001")

# Initialize Local Persistent ChromaDB Vector Database Staging Directory
CHROMA_DIR = "chroma_vector_db"
vector_store = Chroma(
    collection_name="financial_documents",
    embedding_function=embeddings,
    persist_directory=CHROMA_DIR
)

@tool
def ingest_document_text(document_title: str, raw_text: str) -> str:
    """Chunks and embeds raw document text/PDF extractions directly into the local ChromaDB vector store."""
    try:
        # Step 1: Initialize the text splitter (Standard enterprise chunk sizes with overlapping safety)
        text_splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=200)
        chunks = text_splitter.split_text(raw_text)
        
        # Step 2: Generate unique structural IDs for each chunk to prevent duplication
        chunk_ids = [str(uuid.uuid4()) for _ in range(len(chunks))]
        metadata = [{"source": document_title} for _ in chunks]
        
        # Step 3: Write vectors directly to the local disk database
        vector_store.add_texts(texts=chunks, metadatas=metadata, ids=chunk_ids)
        return f"✅ Successfully ingested '{document_title}'. Processed {len(chunks)} text chunks into ChromaDB."
    except Exception as e:
        return f"❌ Failed to ingest document matrix: {str(e)}"

@tool
def query_financial_knowledge_base(user_search_query: str) -> Dict[str, Any]:
    """Queries the local vector database using semantic similarity search to extract relevant context metrics."""
    try:
        # Retrieve the top 3 most semantically aligned data blocks from the database
        docs = vector_store.similarity_search(user_search_query, k=3)
        if not docs:
            return {"status": "empty", "context": "No matching knowledge base documents located."}
        
        extracted_context = "\n---\n".join([f"[Source: {d.metadata.get('source')}]: {d.page_content}" for d in docs])
        return {
            "status": "success",
            "type": "vector_rag_context",
            "context": extracted_context
        }
    except Exception as e:
        return {"error": f"Failed vector query execution: {str(e)}"}
